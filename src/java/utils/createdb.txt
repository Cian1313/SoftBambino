DROP TABLE "USERDATA";

CREATE TABLE "USERDATA"
(
"USER_ID" INT not null primary key
        GENERATED ALWAYS AS IDENTITY
        (START WITH 1, INCREMENT BY 1),   
"EMAIL" VARCHAR(100), 
"PASSWORD" VARCHAR(50), 
"FNAME" VARCHAR(50),   
"LNAME" VARCHAR(50),
"USERTYPE" VARCHAR(10),
"ISDISABLED" BOOLEAN,
"ADDRESSLINEONE" VARCHAR(50),
"ADDRESSLINETWO" VARCHAR(50),
"COUNTY" VARCHAR(25),
"COUNTRY" VARCHAR(70),
"POSTALCODE" VARCHAR(10)
);

DROP TABLE "PRODUCT";

CREATE TABLE "PRODUCT"
(
"PRODUCT_ID" INT not null primary key
        GENERATED ALWAYS AS IDENTITY
        (START WITH 1, INCREMENT BY 1),   
"NAME" VARCHAR(50),
"DESCRIPTION" VARCHAR(100),
"TYPE" VARCHAR(60),
"PRICE" DOUBLE,
"STOCK" INT,
"AGESUITABILITY" VARCHAR(10),
"IMAGE" VARCHAR(100)
);
